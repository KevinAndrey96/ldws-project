  #totalcomputers=@physical.sum(:computers)
  @Sw_status = 1

      @eq=Equipment.where(:etype => "Switch", :power => 3).count

        @Sw_status=0
      else
        @eco=3
      end

    @eco=1
  end

    @servidores+=1

    @servidores+=1

  if @request.proxy?
    @servidores+=1
  end
  if @request.voip?

  end
  if @request.video_conference?

  end
  if @request.active_dir?
    @servidores+=1
  end
  if @request.dhcp?
    @servidores+=1
  end
  if @request.dns?
    @servidores+=1
  end
  if @request.ftp?
    @servidores+=1
  end
  if @request.email?
